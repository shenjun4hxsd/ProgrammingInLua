##载入Lua脚本

####一、通过文件

```csharp
    using UnityEngine;
    using System.Collections;
    using XLua;
    
    public class ByFile : MonoBehaviour {
        LuaEnv luaenv = null;
        // Use this for initialization
        void Start()
        {
            luaenv = new LuaEnv();
            luaenv.DoString("require 'byfile'");
        }
    
        // Update is called once per frame
        void Update()
        {
            if (luaenv != null)
            {
                luaenv.Tick();
            }
        }
    
        void OnDestroy()
        {
            luaenv.Dispose();
        }
    }
```
